CSS Animations



        .animate{
            width: 100px;
            height: 100px;;
            position: relative;
            -webkit-animation-name: photo;
            -webkit-animation-duration: 4s;
            -webkit-animation-iteration-count: infinite;

     @-webkit-keyframes photo {
            0%   {left:0px; top:0px;}

            75%  {left:200px; top:0px;}
            100% {left:100px; top:0px;}
        }

        /* Standard syntax */
        @keyframes photo {
            0%   {left:0px; top:0px;}
            25%  {left:200px; top:0px;}
            50%  {left:300px; top:0px;}
            75%  {left:200px; top:0px;}
            100% {left:100px; top:0px;}
        }
    }

    JavaScript
    function startAnimation(image) {
        document.getElementById(image).className ="animate";


    }
    function stopAnimation(image) {
        document.getElementById(image).className = "";


    }
